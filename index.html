FormLib 简介

FormLib 是一个轻量级的表单元素美化库，提供了比浏览器原生组件更美观、更易用的表单元素。 包含输入框、按钮、对话框、颜色选择器等组件，支持亮色/暗色主题切换，响应式设计适配各种屏幕尺寸。

安装使用

只需引入 CSS 和 JavaScript 文件即可开始使用 FormLib。

步骤 1: 引入 CSS 文件

<link rel="stylesheet" href="path/to/formlib.css">

步骤 2: 引入 JavaScript 文件

<script src="path/to/formlib.js"></script>

步骤 3: 开始使用

FormLib 会在 DOM 加载完成后自动初始化。你可以直接使用预设的 CSS 类名来应用样式。

对话框组件

自定义的确认/取消对话框，支持自定义标题、消息和按钮文本。 

<!-- 提示对话框 -->
<script>
document.getElementById('show-alert').addEventListener('click', function() {
    FormLib.Dialog.alert({
        title: '提示信息',
        message: '这是一个提示对话框，用于显示重要信息。',
        buttonText: '我知道了',
        onClose: function() {
            console.log('对话框已关闭');
        }
    });
});
</script>

<!-- 确认对话框 -->
<script>
document.getElementById('show-confirm').addEventListener('click', function() {
    FormLib.Dialog.confirm({
        title: '确认操作',
        message: '您确定要执行此操作吗？此操作不可撤销。',
        confirmText: '确定',
        cancelText: '取消',
        onConfirm: function() {
            console.log('用户点击了确定');
            FormLib.Dialog.alert({
                title: '操作成功',
                message: '您的操作已成功执行！'
            });
        },
        onCancel: function() {
            console.log('用户点击了取消');
        }
    });
});
</script>

<!-- 简单版颜色选择器 -->
<div id="colorpicker-simple" data-formlib="colorpicker" data-default-color="#3b82f6"></div>

<!-- 专业版颜色选择器 -->
<div id="colorpicker-professional"></div>

<script>
// 简单版颜色选择器（自动初始化）
document.addEventListener('DOMContentLoaded', function() {
    // 手动初始化专业版颜色选择器
    const professionalColorpicker = FormLib.ColorPicker.init(
        document.getElementById('colorpicker-professional'),
        {
            mode: 'professional',
            defaultColor: '#3b82f6',
            onChange: function(color, isCustom) {
                console.log('专业版选择的颜色:', color);
                console.log('是否自定义颜色:', isCustom);
                
                // 更新显示
                document.getElementById('color-value-professional').textContent = '当前颜色: ' + color.toUpperCase();
                document.getElementById('preview-professional').style.backgroundColor = color;
            }
        }
    );
    
    // 简单版颜色选择器事件监听
    const simpleColorpicker = FormLib.ColorPicker.init(
        document.getElementById('colorpicker-simple'),
        {
            mode: 'simple',
            defaultColor: '#3b82f6',
            onChange: function(color, isCustom) {
                console.log('简单版选择的颜色:', color);
                console.log('是否自定义颜色:', isCustom);
                
                // 更新显示
                document.getElementById('color-value-simple').textContent = '当前颜色: ' + color.toUpperCase();
                document.getElementById('preview-simple').style.backgroundColor = color;
            }
        }
    );
    
    // 应用简单版颜色按钮
    document.getElementById('apply-color-simple').addEventListener('click', function() {
        const color = simpleColorpicker.getColor();
        
        FormLib.Dialog.alert({
            title: '颜色已应用',
            message: '您选择的简单版颜色 ' + color + ' 已应用！',
            buttonText: '太棒了'
        });
    });
    
    // 应用专业版颜色按钮
    document.getElementById('apply-color-professional').addEventListener('click', function() {
        const color = professionalColorpicker.getColor();
        
        FormLib.Dialog.alert({
            title: '颜色已应用',
            message: '您选择的专业版颜色 ' + color + ' 已应用！',
            buttonText: '太棒了'
        });
    });
    
    // 对比颜色按钮
    document.getElementById('compare-colors').addEventListener('click', function() {
        const simpleColor = simpleColorpicker.getColor();
        const professionalColor = professionalColorpicker.getColor();
        
        FormLib.Dialog.confirm({
            title: '颜色对比',
            message: `简单版颜色: ${simpleColor}
专业版颜色: ${professionalColor}

两个颜色${simpleColor === professionalColor ? '相同' : '不同'}`,
            confirmText: '确定',
            cancelText: '取消'
        });
    });
});
</script>

表单验证

强大的表单验证功能，支持各种验证规则和实时反馈。 

<!-- HTML -->
<form id="validation-form" data-formlib="form">
    <div class="formlib-input-group">
        <label class="formlib-input-label" for="username">用户名</label>
        <input 
            type="text" 
            id="username" 
            name="username"
            class="formlib-input" 
            data-required
            data-minlength="3"
            data-maxlength="20"
        >
    </div>
    
    <button type="submit" class="formlib-btn formlib-btn-primary">提交表单</button>
</form>

<!-- JavaScript -->
<script>
// 手动初始化表单验证
const validator = FormLib.FormValidator.init(document.getElementById('validation-form'), {
    validateOnBlur: true,  // 失去焦点时验证
    validateOnChange: false, // 值改变时验证
    onSubmit: function(isValid, form, fields) {
        console.log('表单提交', { isValid, form, fields });
    },
    onSuccess: function(form) {
        console.log('表单验证成功，可以提交数据');
        
        // 模拟表单提交
        FormLib.Dialog.alert({
            title: '提交成功',
            message: '您的表单已成功提交！',
            onClose: function() {
                form.reset();
            }
        });
    },
    onError: function(form, fields) {
        console.log('表单验证失败', fields);
    }
});
</script>

主题系统

支持亮色和暗色主题，可自动根据系统偏好设置，也可手动切换。

当前主题信息

当前主题: 亮色

系统偏好: 检测中...

<!-- 主题切换按钮 -->
<button id="theme-toggle" data-formlib="theme-toggle">
    <i class="fa fa-moon-o dark:hidden"></i>
    <i class="fa fa-sun-o hidden dark:block"></i>
</button>

<!-- JavaScript -->
<script>
// 设置亮色主题
document.getElementById('theme-light').addEventListener('click', function() {
    FormLib.Theme.setTheme('light');
    updateThemeInfo();
});

// 设置暗色主题
document.getElementById('theme-dark').addEventListener('click', function() {
    FormLib.Theme.setTheme('dark');
    updateThemeInfo();
});

// 跟随系统
document.getElementById('theme-auto').addEventListener('click', function() {
    localStorage.removeItem('formlib-theme');
    FormLib.Theme.init();
    updateThemeInfo();
});

// 更新主题信息显示
function updateThemeInfo() {
    const currentTheme = FormLib.Theme.getTheme();
    const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    
    document.getElementById('current-theme').textContent = currentTheme === 'dark' ? '暗色' : '亮色';
    document.getElementById('system-theme').textContent = systemPrefersDark ? '暗色' : '亮色';
}

// 监听系统主题变化
window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', function(e) {
    if (!localStorage.getItem('formlib-theme')) {
        FormLib.Theme.init();
        updateThemeInfo();
    }
});

// 初始化
updateThemeInfo();
</script>
